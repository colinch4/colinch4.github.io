const { ApolloServer, gql } = require('apollo-server');

const db = {
  books: [
    { title: 'The Awakening', author: 'Kate Chopin' },
    { title: 'City of Glass', author: 'Paul Auster' },
  ]
};

const typeDefs = gql`
  type Book {
    title: String
    author: String
  }

  type Query {
    books: [Book]
  }
`;


const resolvers = {
  Query: {
    books: () => db.books,
  },
};


const server = new ApolloServer({ typeDefs, resolvers });
