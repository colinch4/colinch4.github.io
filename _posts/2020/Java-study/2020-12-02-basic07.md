  for(int i = 0, i <= 100, i++>){
    if(isPrimeNumber(i)) // 
      System.out.println(i);
  }
}
public static boolean isPrimeNumber  (int num) {          // boolean 타입이 오면 함수명 is 붙이고 시작
  if(num == 1) {
    return false;
  }
  for(int j = 2, 2 < num; j++) {
    if (num % j == 0) {
      return false;
    }
    return true;  // return 종료의 의미
  }
  
}
